#$ESTATE$: Estate to check influence for
#$EVENTTARGET$: Event target string used for a random province granted to the $ESTATE$

		random_owned_province = {

		limit = {

		}

			type = $ADVISORTYPE$
			skill = 3

add_or_spawn_center_of_trade = {
	if = { #If has center of trade, add level.
		limit = {
			OR = {
				province_has_center_of_trade_of_level = 1
				province_has_center_of_trade_of_level = 2
				province_has_center_of_trade_of_level = 3
			}
		}
		add_center_of_trade_level = $LEVEL$
	}
	else = {
		center_of_trade = $LEVEL$
	}
}

create_interregnum_effect = {
	hidden_effect = {
		every_country = {
			limit = {
				marriage_with = ROOT
			}
			break_marriage = ROOT
		}
		kill_heir = yes
	}
	define_ruler = {
		name = "(Interregnum)"
		adm = 0
		dip = 0
		mil = 0
		fixed = yes
		regency = yes
	}
}

#Same version as vanilla's move_capital_effect, but using a custom TAG instead of ROOT for the country in question. Important for subject country formations used in this mod.
#THIS = Province Scope, TAG = Country Scope
#Usage: TAG_move_capital_effect = { TAG=XXX } inside a province scope to move the tag's capital to.
TAG_move_capital_effect = {
	if = {
		limit = {
			is_part_of_hre = yes
			$TAG$ = {
				is_part_of_hre = no
			}
		}
		set_in_empire = no
		emperor = {
			add_opinion = {
				who = $TAG$
				modifier = hre_province_left
			}
		}
		add_imperial_influence = -1
	}
	if = {
		limit = { has_estate = yes }
		remove_estate = yes
	}
	if = {
		limit = { is_capital = no }
		$TAG$ = {
			set_capital = PREV
		}
	}
}
