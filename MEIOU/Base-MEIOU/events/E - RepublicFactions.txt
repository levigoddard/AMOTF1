province_event = {
	id = republic_factions.1
	title = "republic_factions.1.t"
	desc = "republic_factions.1.d"
	picture = DEBATE_REPUBLICAN_eventPicture
	is_mtth_scaled_to_size = 1
	trigger = {
		has_dlc = "Res Publica"
		owner = {
			government = republic
			OR = {
				AND = {
					OR = {
						has_reform = city_state
						AND = {
							NOT = {
								has_reform = free_city
							}
							NOT = {
								has_reform = trade_republic
							}
							NOT = {
								has_reform = colonial_government
							}
							NOT = {
								has_reform = dictator_reform
							}
							has_reform = oligarchy_reform
						}
						has_reform = oligarchic_republic
						has_reform = ambrosian_republic_legacy
					}
				}
				AND = {
					OR = {
						has_reform = trade_republic
						has_reform = merchant_republic
						has_reform = venetian_merchant_republic
					}
				}
			}
		}
		NOT = {
			has_province_modifier = free_trade_of_goods
		}
		NOT = {
			has_province_modifier = strong_guild
		}
	}
	mean_time_to_happen = {
		years = 66
	}
	option = {
		name = "republic_factions.1.a"
		owner = {
			add_faction_influence = {
				faction = mr_traders
				influence = 5
			}
			add_faction_influence = {
				faction = mr_guilds
				influence = -5
			}
			every_owned_province = {
				limit = {
					trade_goods = ROOT
				}
				add_province_modifier = {
					name = "free_trade_of_goods"
					duration = 1095
				}
			}
		}
	}
	option = {
		name = "republic_factions.1.b"
		owner = {
			add_faction_influence = {
				faction = mr_guilds
				influence = 5
			}
			add_faction_influence = {
				faction = mr_traders
				influence = -5
			}
