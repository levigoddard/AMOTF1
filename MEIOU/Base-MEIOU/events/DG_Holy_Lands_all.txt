country_event = {
	id = dg_holylands.001

		has_country_flag = holy_land_jerusalem

			religion_group = jewish_group

				religion_group = muslim

					religion = druze
				}
			}
		}
		piety = -0.60
	}
	mean_time_to_happen = {
		months = 3
	}
	immediate = {
		clr_country_flag = holy_land_jerusalem
		if = {
			limit = {
				religion_group = jewish_group
			}
			hidden_effect = {
				change_variable = {
					which = jerusalem
					value = 1
				}
			}
		}
		if = {
			limit = {
				379 = {
					is_core = ROOT
				}
			}
			hidden_effect = {
				change_variable = {
					which = jerusalem
					value = 1
				}
			}
		}
		if = {
			limit = {
				379 = {
					owner = {
						NOT = {
							religion_group = ROOT
						}
					}
				}
			}
			hidden_effect = {
				change_variable = {
					which = jerusalem
					value = 1
				}
			}
			custom_tooltip = lost_holyland_heathen
		}
		if = {
			limit = {
				379 = {
					owner = {
						religion = ROOT
					}
				}
			}
			hidden_effect = {
				change_variable = {
					which = jerusalem

			}
			custom_tooltip = lost_holyland_coreligionist

					which = jerusalem

			}
			if = {

						NOT = {
							has_country_modifier = religious_scandal_country_strong
						}

							has_country_modifier = religious_scandal_country
						}
						NOT = {
							has_country_modifier = religious_scandal_country_weak
						}
					}
				}

					}

						name = religious_scandal_country

					}
				}
				else = {
					add_country_modifier = {
