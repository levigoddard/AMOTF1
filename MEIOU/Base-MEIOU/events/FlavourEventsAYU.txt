			}
		}

		name = "flavor_ayu.EVTOPTA3"

			30 = {}
			40 = {
				custom_tooltip = stability_points_lost_50

					subtract_variable = {
						which = stability_points
						value = 50
					}

						limit = {
							NOT = {
								check_variable = {
									which = stability_points
									value = 0
								}
							}
						}

							id = POP_Stability.003
						}

							add_mandate = -5
						}
					}
				}
			}
			30 = {
				add_stability = -1

					country_event = {
						id = POP_Stability.003
					}
				}
				if = {
					limit = {
						is_emperor_of_china = yes
					}
					add_mandate = -5
				}
			}
		}
		add_mercantilism = 15
		add_inflation = -5
		random_owned_province = {
			limit = {
				has_province_flag = AYU_eco_reform_flag1
			}
			add_province_modifier = {
				name = "industrious_ruler_gravity"
				duration = 3650
			}
			clr_province_flag = AYU_eco_reform_flag1
		}
		random_owned_province = {
			limit = {
				has_province_flag = AYU_eco_reform_flag2
			}
			add_province_modifier = {
				name = "industrious_ruler_gravity"

			}

		}
