on_religion_change = {
	if = {
		limit = {
			has_heir = yes
		}
		set_heir_religion = ROOT
	}
	if = {
		limit = {
			has_regency = no
		}
		set_ruler_religion = ROOT
	}
	if = {
		limit = {
			has_dlc = "Rights of Man"
			has_consort = yes
		}
		set_consort_religion = ROOT
	}
	if = {
		limit = {
			always = yes
		}
		set_country_flag = changed_state_religion
		every_owned_province = {
			limit = {
				is_city = yes
				has_building = temple_heathen
			}
			if = {
				limit = {
					has_owner_religion = yes
				}
				remove_building = temple_heathen
				if = {
					limit = {
						NOT = {
							has_building = temple
						}
					}
					add_building = temple
				}
			}
			else = {
				set_province_flag = temple_to_convert
				owner = {
					if = {
						limit = {
							NOT = {
								has_country_flag = convert_heathen_temples
							}
						}
						set_country_flag = convert_heathen_temples
						country_event = {
							id = dg_piety.056
							days = 2
						}
					}
				}
			}
		}
		country_event = {
			id = dg_conversion.003
			days = 0
		}
	}
	if = {
		limit = {
			OR = {
				tag = PAP
				tag = AVI
			}
			NOT = {
				religion = catholic
			}
		}
		change_religion = catholic
	}
	events = { propagate_religion_events.7 }
}
