country_event = { #1 Shattered [Root.GetExarchesName]
	id = exarches_estate_events.1
	title = "exarches_estate_events.1.t" 
	desc = "exarches_estate_events.1.d"
	picture = DIET_eventPicture
	
	trigger = {
		
		has_country_modifier = the_exarch_system
		OR = {
			has_estate = estate_exarches
			any_country = {
				is_exarch_tag_and_exarch_of_root_trigger = yes
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_exarches
				modifier = EST_VAL_EXARCH_LOWERED_MERCANTILISM
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_exarches
				modifier = EST_VAL_EXARCH_NOT_LOWERED_MERCANTILISM
			}
		}
		mercantilism = 10
	}

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_exarches
				influence = 50
			}
		}
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_exarches
				influence = 70
			}
		}
		modifier = {
			factor = 2.0
			mercantilism = 25
		}
		modifier = {
			factor = 2.0
			mercantilism = 50
		}
		modifier = {
			factor = 2.0
			mercantilism = 75
		}
	}
	
	option = { #Lower mercantilism
		name = "exarches_estate_events.1.a"
		ai_chance = { 
			factor = 50
		}
		
		add_mercantilism = -1
		if = {
			limit = {
				has_estate = estate_exarches
			}
			add_estate_loyalty = {
				estate = estate_exarches
				loyalty = 15
			}
			add_estate_influence_modifier = {
				estate = estate_strategoi
				desc = EST_VAL_EXARCH_LOWERED_MERCANTILISM
				influence = 10
				duration = 3650
			}
		}
		if = {
			limit = {
				any_country = {
					is_exarch_tag_and_exarch_of_root_trigger = yes
				}
			}
			random_country = {
				limit = {
					is_exarch_tag_and_exarch_of_root_trigger = yes
				}
				add_liberty_desire = -15
			}
		}
		
	}
	option = { #Do not
		name = "exarches_estate_events.1.b"
		ai_chance = { 
			factor = 50
		}
		
		if = {
			limit = {
				has_estate = estate_exarches
			}
			add_estate_loyalty = {
				estate = estate_exarches
				loyalty = -10
			}
			add_estate_influence_modifier = {
				estate = estate_strategoi
				desc = EST_VAL_EXARCH_NOT_LOWERED_MERCANTILISM
				influence = -10
				duration = 3650
			}
		}
		if = {
			limit = {
				any_country = {
					is_exarch_tag_and_exarch_of_root_trigger = yes
				}
			}
			every_country = {
				limit = {
					is_exarch_tag_and_exarch_of_root_trigger = yes
				}
				add_liberty_desire = 10
			}
		}
	}
}
