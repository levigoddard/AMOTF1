# Vassal upkeep bonus
###########################################
bt_triggered_modifier_a_big_family = {
	potential = {
		num_of_subjects = 1
		NOT = {
			calc_true_if = {
				all_known_country = {
					is_subject_of = ROOT
					OR = {
						is_subject_of_type = vassal
						is_subject_of_type = march
						is_subject_of_type = personal_union
						is_subject_of_type = client_vassal
						is_subject_of_type = client_march
					}				
				}
				amount = 5
			}
		}
	}
	trigger = {
		custom_trigger_tooltip = {
			tooltip = bt_bunch_o_vassals_tooltip
			calc_true_if = {
				all_known_country = {
					is_subject_of = ROOT
					OR = {
						is_subject_of_type = vassal
						is_subject_of_type = march
						is_subject_of_type = personal_union
						is_subject_of_type = client_vassal
						is_subject_of_type = client_march
					}					
				}
				amount = 3
			}
		}
		hidden_trigger = {
			NOT = {
				calc_true_if = {
					all_known_country = {
						is_subject_of = ROOT
						OR = {
							is_subject_of_type = vassal
							is_subject_of_type = march
							is_subject_of_type = personal_union
							is_subject_of_type = client_vassal
							is_subject_of_type = client_march
						}				
					}
					amount = 5
				}
			}
		}
	}
	diplomatic_upkeep = 1
	reduced_liberty_desire = 10
}

bt_iberian_crowns = {
	potential = {
		tag = ARA
		is_subject = no
		NOT = { total_development = 500 }
		any_subject_country = {
			is_subject_of_type = personal_union
			OR = {
				capital_scope = { region = italy_region }
				capital_scope = { region = iberia_region }
			}
		}
	}
	trigger = {
		tag = ARA
		any_subject_country = {
			is_subject_of_type = personal_union
			total_development = 200
		}
		#NOT = { has_country_flag = bt_centralising_aragon }			#Currently a dead end, might add later.
	}
	reduced_liberty_desire = 30
	diplomatic_annexation_cost = 0.25
}
